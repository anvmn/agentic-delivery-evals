function healthstats_permission() {

      'description' => t('Access the JSON health statistics endpoint.'),
    ),
  );

/**
 * Implements hook_menu().
 */
function healthstats_menu() {
  $items['api/healthstats'] = array(
    'title' => 'Health stats',
    'page callback' => 'healthstats_page',
    'access arguments' => array('view healthstats'),
    'delivery callback' => 'healthstats_deliver_json',
    'type' => MENU_CALLBACK,
  );
  return $items;
}

/**
 * Page callback for /api/healthstats.
 *
 * @return array
 *   Associative array of health statistics.
 */
function healthstats_page() {
  $users = db_query('SELECT COUNT(uid) FROM {users} WHERE status = 1')->fetchField();
  $nodes = db_query('SELECT COUNT(nid) FROM {node} WHERE status = 1')->fetchField();

  return array(
    'users' => (int) $users,
    'nodes' => (int) $nodes,
  );
}

/**
 * Delivery callback for /api/healthstats.
 *
 * Encodes the page callback result as JSON for authorized users, and falls back
 * to Drupal's normal access-denied / not-found handling otherwise.
 *
 * @param mixed $page_callback_result
 *   The value returned by the page callback or a menu status constant.
 */
function healthstats_deliver_json($page_callback_result) {
  if ($page_callback_result == MENU_ACCESS_DENIED) {
    drupal_access_denied();
  }
  elseif ($page_callback_result == MENU_NOT_FOUND) {
    drupal_not_found();
  }
  else {
    drupal_json_output($page_callback_result);
  }
}
